arr = [3.4, 5.1, 2.6, 1.3, 7.8, 4.2, 6.5]

min_val = min(arr)  # минимальное значение в массиве
max_val = max(arr)  # максимальное значение в массиве

if max_val - min_val <= 25:
    print("Максимальный элемент превышает минимальный не более чем на 25")
else:
    print("Максимальный элемент больше минимального на более чем 25")

if min_val * 2 < max_val:
    print("Минимальный элемент меньше максимального более чем в 2 раза")
else:
    print("Минимальный элемент не меньше максимального более чем в 2 раза")

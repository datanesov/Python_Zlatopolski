arr = [] # Создание пустого списка
num = int(input("Введите число: ")) # Ввод первого числа
i = 0 # Начальный индекс

while num != -1: # Чтение чисел до тех пор, пока не введен -1
    arr.append(num) # Добавление числа в список
    num = int(input("Введите число: ")) # Чтение следующего числа
    i += 1 # Приращение индекса

j = 0 # Начальный индекс для цикла while

while j < len(arr) - 1: # Перебор списка с помощью цикла while
    if arr[j] == arr[j+1]: # Если текущее число равно следующему
        print("Пара одинаковых соседних чисел найдена!")
        print("Их порядковые номера:", j+1, "и", j+2) # Вывод порядковых номеров первой из таких пар
        break # Прерывание цикла после нахождения первой пары
    j += 1 # Приращение индекса

# Если цикл while завершился без нахождения пары чисел
else:
    print("Пара одинаковых соседних чисел не найдена.")

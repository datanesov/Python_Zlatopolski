n = int(input("Введите количество секунд: "))

# a) полных часов с начала суток
hours = n // 3600
print("Полных часов с начала суток:", hours)

# б) полных минут прошло с начала очередного часа
minutes = (n % 3600) // 60  # остаток после часов, деленный на 60
print("Полных минут с начала очередного часа:", minutes)

# в) полных секунд прошло с начала очередной минуты
seconds = (n % 3600) % 60  # остаток после часов и минут
print("Полных секунд с начала очередной минуты:", seconds)

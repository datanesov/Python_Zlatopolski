exam1 = [5, 4, 4, 2, 3, 5, 4, 5, 3, 2]
exam2 = [3, 4, 5, 2, 4, 3, 2, 5, 2, 4]

# Используем цикл для подсчета количества студентов с двойкой на обоих экзаменах
count = 0
for i in range(len(exam1)):
    if exam1[i] == 2 and exam2[i] == 2:
        count += 1

print("Количество студентов, получивших на обоих экзаменах двойку:", count)

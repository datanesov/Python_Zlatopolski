array = [] # создаем пустой массив
i = 1 # начальное значение
while i <= 25: # пока не достигнем 25
    array.insert(i-1,i) # добавляем элемент в массив с индексом i-1
    i += 1 # увеличиваем счетчик
print(array)
array.insert(-1,100) # добавляем 100 в конец массива
array.insert(-1,200) # добавляем 200 в конец массива
print(array)

points = [30, 25, 19, 14, 32, 28, 21, 12] # список очков команд
min_points = points[0] # сначала минимальное значение равно первому элементу списка
min_index = 0 # создаем переменную для хранения индекса команды с минимальным количеством очков
for i in range(1, len(points)): # проходимся по оставшимся элементам списка
    if points[i] < min_points: # если очки команды i меньше минимального значения
        min_points = points[i] # обновляем минимальное значение
        min_index = i # обновляем индекс команды с минимальным количеством очков
print("Команда №", min_index + 1, "набрала наименьшее количество очков:", min_points)

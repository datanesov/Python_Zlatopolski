array = [64, 600, 3, 1234, 65634, 42, 65000, 98765, 200000]

# а) поиск первого элемента, большего 65530
for i in range(len(array)):
    if array[i] > 65530:
        print("Номер первого элемента, большего 65530: ", i)
        break

# б) поиск последнего элемента, большего 65530
for i in range(len(array)-1, -1, -1):
    if array[i] > 65530:
        print("Номер последнего элемента, большего 65530: ", i)
        break

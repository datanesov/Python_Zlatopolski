# б) Меняем местами соседние элементы с помощью цикла:

a = [1, 2, 3, 4, 5, 6]
for i in range(0, len(a)-1, 2):
    a[i], a[i+1] = a[i+1], a[i]
print(a)

# в) Меняем местами элементы из начала и конца списка с помощью цикла:

a = [1, 2, 3, 4, 5, 6]
for i in range(len(a)//2):
    a[i], a[-i-1] = a[-i-1], a[i]
print(a)

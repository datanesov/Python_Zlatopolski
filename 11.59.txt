arr = [10, 15, 25, 8, 20, 18, 22, 5]
sum_a = 0
sum_b = 0
a = 15

# считаем сумму элементов, значение которых не превышает 20
for i in range(len(arr)):
    if arr[i] <= 20:
        sum_a += arr[i]

# считаем сумму элементов, больших числа а
for i in range(len(arr)):
    if arr[i] > a:
        sum_b += arr[i]

print("Сумма элементов массива, значение которых не превышает 20: ", sum_a)
print("Сумма элементов массива, больших числа а: ", sum_b)

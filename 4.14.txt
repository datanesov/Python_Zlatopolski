r1 = float(input("Введите сопротивление первого участка: "))
u1 = float(input("Введите напряжение на первом участке: "))
r2 = float(input("Введите сопротивление второго участка: "))
u2 = float(input("Введите напряжение на втором участке: "))

i1 = u1 / r1
i2 = u2 / r2

if i1 > i2:
    print("Меньший ток протекает по второму участку")
elif i1 < i2:
    print("Меньший ток протекает по первому участку")
else:
    print("Токи равны")

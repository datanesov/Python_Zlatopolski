# оценки по геометрии
marks = [5, 5, 5, 5, 5, 4, 3, 2, 1, 4, 3, 2, 3, 4, 5, 4, 3, 2, 1, 2, 3, 4, 5, 4, 3]

# счетчик учеников с оценкой 5
num_of_fives = 0

# проходим по списку оценок и увеличиваем счетчик, если оценка равна 5
for mark in marks:
  if mark == 5:
    num_of_fives += 1

print("Количество учеников с оценкой '5':", num_of_fives)

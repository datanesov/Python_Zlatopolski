
# Решение через список и индексацию

cards = ['шестерка', 'семерка', 'восьмерка', 'девятка',
         'десятка', 'валет', 'дама', 'король', 'туз']

k = int(input('Введите номер карты от 7 до 14: '))
if 6 < k <= 14:
    index = k - 7
    print(f'Достоинство карты: {cards[index]}')
else:
    print('Некорректный номер карты')

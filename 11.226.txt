import numpy as np

areas = np.array([10, 12, 15, 8, 13, 20, 18, 16, 11, 9]) # площади засеваемые пшеницей в каждом районе
yields = np.array([40, 35, 38, 45, 42, 50, 48, 46, 39, 41]) # средняя урожайность в каждом районе в центнерах с гектара

total_wheat = np.sum(areas * yields)
avg_yield = np.mean(yields)

print("Количество пшеницы, собранное в области:", total_wheat, "центнеров")
print("Средняя урожайность по области:", avg_yield, "центнеров с гектара")

a = 'how'
b = 'are'
c = 'you'
# меняем значения переменных местами
a = a + b + c # объединяем значения в одну строку
b = a[:len(b)] # берем первую часть строки для b
c = a[len(b):len(a)-len(c)] # берем среднюю часть строки для c
a = a[len(b)+len(c):] # берем последнюю часть строки для a

print(a) # 'you'
print(b) # 'how'
print(c) # 'are'

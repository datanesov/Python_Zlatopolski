b = [1, 5, 8, 21, 22, 26, 36, 40, 46, 50, 56, 60, 70, 80]
summa = sum(filter(lambda x: x > 20, b))

if summa > 100:
    print('Сумма чисел, больших 20, превышает 100')
else:
    print('Сумма чисел, больших 20, не превышает 100')

b = [1, 5, 8, 21, 22, 26, 36, 40, 46, 50, 56, 60, 70, 80]
summa = sum(filter(lambda x: x < 50, b))

if summa % 2 == 0:
    print('Сумма чисел, меньших 50, является четным числом')
else:
    print('Сумма чисел, меньших 50, не является четным числом')


m = int(input("Введите номер месяца: "))
n = int(input("Введите число: "))

if n == 1:
  if m == 1:
    print("Данный день не определяет 1 января")
  else:
    m -= 1
    n = 31
elif n == 31:
  if m == 12:
    print("Данный день не определяет 31 декабря")
  else:
    m += 1
    n = 1
else:
  n -= 1
print("Дата предыдущего дня:", m, n)

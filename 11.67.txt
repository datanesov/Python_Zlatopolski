res1 = [0, 0] # res1[0] - кол-во жителей на нечетной стороне, res1[1] - на четной
houses = [5, 7, 10, 3, 8, 12, 6, 4, 9] # список кол-ва жителей в домах

for i in range(len(houses)):
    if i % 2 == 0: # четный номер дома
        res1[1] += houses[i]
    else: # нечетный номер дома
        res1[0] += houses[i]

if res1[0] > res1[1]:
    print("Больше жителей проживает на нечетной стороне улицы")
else:
    print("Больше жителей проживает на четной стороне улицы")

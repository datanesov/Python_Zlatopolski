# a) Условие для треугольной фигуры:
def check_triangle(x, y):
    if ((x >= -2 and y+2*x <= 5) or (x <= -2 and y <= 1)): 
        print("Точка находится в заштрихованной области")
    else:
        print("Точка не находится в заштрихованной области")

# б) Условие для прямоугольной фигуры:
def check_rectangle(x, y):
    if ((y == 1.5 and x >= -float('inf') and x <= float('inf')) or 
        (y == -2 and x >= -float('inf') and x <= float('inf')) or 
        (y <= 1.5 and y >= -2 and x >= -float('inf') and x <= float('inf'))): 
        print("Точка находится в заштрихованной области")
    else:
        print("Точка не находится в заштрихованной области")

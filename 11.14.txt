n = int(input("Введите натуральное число: "))
s = str(n) # преобразуем число в строку
lst = [0] * 6 # создаем список из 6 элементов, незаполненные значения равны нулю
for i, digit in enumerate(s[::-1]): # проходим по строке в обратном порядке, используя функцию enumerate, которая позволяет получить индексы элементов
    lst[i] = int(digit) # преобразуем цифру из строки в целое число и записываем в список
print(lst)
for j in range(len(lst)):
    if lst[j] != 0:
        print(lst[j])

k = int(input("Введите номер дня в году (1 < k < 365): "))
d = int(input("Введите номер дня недели, на который приходится 1 января (1 - понедельник, 2 - вторник и т.д.): "))

# вычисляем номер дня недели для k-го дня года
day_of_week = (k + d - 2) % 7 + 1  # -2, т.к. 1 января уже учтено, +1, т.к. номер дня недели от 1 до 7

# определяем значение p в зависимости от дня недели
if day_of_week <= 5:  # с понедельника по пятницу
    p = day_of_week
else:  # суббота или воскресенье
    p = 0

print("В", k, "день года:", end=' ')
if p == 0:
    print("выходной")
else:
    print(p, "-й день недели")

age = int(input("Введите возраст: ")) # принимаем возраст от пользователя

ending = {1: "год", 2: "года", 3: "года", 4: "года"} # создаём словарь окончаний для возраста оканчивающегося на 1, 2, 3 или 4

if age % 10 in ending.keys() and age % 100 not in [11, 12, 13, 14]: # проверяем, что возраст оканчивается на 1, 2, 3 или 4 (но не на числа типа 11, 12, 13, 14, 111, 112, 113, 114, и т.д.)
    print("Мне", age, ending[age % 10])
else: # если возраст оканчивается на другую цифру
    print("Мне", age, "лет")

a = float(input("Введите первое число: "))
b = float(input("Введите второе число: "))

min_num = a if a < b else b  # первый условный оператор для определения минимального числа
max_num = a if a > b else b  # второй условный оператор для определения максимального числа

print("Минимальное число:", min_num)
print("Максимальное число:", max_num)

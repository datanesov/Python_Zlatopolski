items1 = [10, 20, 30, 40, 50, 60, 70, 80] # стоимости первого набора предметов
items2 = [15, 25, 35, 45, 55, 65, 75, 85] # стоимости второго набора предметов

avg1 = sum(items1) / len(items1) # средняя стоимость предмета в первом наборе
avg2 = sum(items2) / len(items2) # средняя стоимость предмета во втором наборе

if avg1 < avg2:
    print("Первый набор предметов дешевле")
elif avg1 > avg2:
    print("Второй набор предметов дешевле")
else:
    print("Наборы предметов стоят одинаково")

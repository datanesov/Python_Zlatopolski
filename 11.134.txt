ages = [25, 30, 40, 20, 30, 40, 50, 60]

# Получаем минимальный и максимальный возраст из списка
min_age = min(ages)
max_age = max(ages)

# Если первый элемент списка - самый старший возраст, то указан раньше
if ages[0] == max_age:
    print("Самый старший возраст указан раньше")
else:
    print("Самый младший возраст указан раньше")

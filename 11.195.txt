arr = [3, 3, 3, 5, 6, 7, 8, 9]
count = 1
n = len(arr)

# ищем количество одинаковых элементов в начале массива
while count < n and arr[count] == arr[0]:
    count += 1

# выводим все элементы, следующие за последним из равных элементов
if count < n:
    print("Количество равных элементов в начале массива:", count)
    print("Следующие за последним из равных элементы:", arr[count:])
else:
    print("Все элементы равны в начале массива")

heights = [165, 170, 175, 180, 160, 165, 170, 185, 170, 175,
           170, 170, 165, 170, 172, 170, 175, 180, 172, 169,
           173, 176, 171, 175, 177, 170, 174, 170, 168, 180]
count = 0
for height in heights:
    if height > 170:
        count += 1
print("Количество учеников с ростом больше 170 см:", count)

if count >= 5:
    print("Можно сформировать баскетбольную команду")
else:
    print("Нельзя сформировать баскетбольную команду")

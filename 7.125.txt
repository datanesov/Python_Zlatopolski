y = [1.2, 2.3, 4.5, 6.7, 8.9, 9.0, 10.1, 12.3, 13.4, 14.5, 15.6, 16.7, 18.9, 20.0, 21.1]
n = 11.5

# Сумма чисел меньших n
sum_less_n = 0
for y_i in y:
    if y_i < n:
        sum_less_n += y_i
print(f"Сумма чисел меньших {n} равна {sum_less_n}")

# Два элемента последовательности в интервале между которыми находится значение n
for i in range(len(y)-1):
    if y[i] < n < y[i+1]:
        print(f"Элементы на позициях {i} и {i+1} равны соответственно {y[i]} и {y[i+1]}")
        break

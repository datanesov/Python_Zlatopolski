num = input() # ввод числа как строки
sum = 0 # переменная для суммы

for i in range(len(num)):
    if i == 0: # для первой цифры
        sum += int(num[i]) # прибавляем цифру
    elif i == len(num)-1: # для последней цифры
        sum += ((-1)**i) * int(num[i]) # прибавляем с учетом знака
    else:
        sum += ((-1)**i) * int(num[i]) # прибавляем с учетом знака

print(sum) # выводим результат

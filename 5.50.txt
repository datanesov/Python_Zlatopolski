a = 10  # начальный пробег в км
p = 0.1  # процент увеличения пробега на 10%

# пробег за второй, третий, ..., десятый день тренировок
an = a
for n in range(2, 11):
    an = an + (an * p)
    print(f"Пробег за {n}-й день тренировок: {an:.2f} км")

# суммарный пробег за первые 7 дней тренировок
sn = 0
an = a
for n in range(1, 8):
    sn = sn + an
    an = an + (an * p)
print(f"Суммарный пробег за первые 7 дней тренировок: {sn:.2f} км")

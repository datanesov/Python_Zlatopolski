x1, y1, w1, h1 = map(int, input('Введите координаты и размеры первого прямоугольника (через пробел): ').split())
x2, y2, w2, h2 = map(int, input('Введите координаты и размеры второго прямоугольника (через пробел): ').split())

# Находим координаты углов каждого прямоугольника
x1_min, y1_min, x1_max, y1_max = x1, y1, x1 + w1, y1 + h1
x2_min, y2_min, x2_max, y2_max = x2, y2, x2 + w2, y2 + h2

# Находим координаты левого нижнего и правого верхнего углов минимального прямоугольника
x_min = min(x1_min, x2_min)
y_min = min(y1_min, y2_min)
x_max = max(x1_max, x2_max)
y_max = max(y1_max, y2_max)

# Выводим результат
print(f'Координаты левого нижнего угла минимального прямоугольника: ({x_min}, {y_min})')
print(f'Координаты правого верхнего угла минимального прямоугольника: ({x_max}, {y_max})')

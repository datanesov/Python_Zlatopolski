grades = [3, 4, 2, 5, 3, 4, 2, 4, 3, 4, 3, 2, 5, 3, 4, 4, 2, 3, 4, 3, 3, 4, 2, 5, 4]

# счетчик неуспевающих
count = 0

# проходим по всем элементам массива
for grade in grades:
  # проверяем условие неуспевающей оценки
  if grade < 3:
    count += 1
print("Количество неуспевающих учеников по химии:", count)

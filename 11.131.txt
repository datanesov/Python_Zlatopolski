weights = [45, 62, 39, 56, 81, 70]

min_weight = min(weights)  # минимальный вес в группе
max_weight = max(weights)  # максимальный вес в группе

if max_weight > min_weight * 2:
    print("Вес самого тяжелого человека превышает массу самого легкого более чем в 2 раза")
else:
    print("Вес самого тяжелого человека не превышает массу самого легкого более чем в 2 раза")

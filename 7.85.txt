x = [1.2, 2.5, 33.2, 36.7, 25.1, 33.2, 16.3, 9.4, 33.2, 17.6, 13.8, 28.5, 33.2, 19.7, 7.8]
filtered = list(filter(lambda i: i <= 33.2 and i % 4 == 0, x))
count = len(filtered)
if count == 0:
    print("Количество чисел не кратно четырем")
else:
    print("Количество чисел, не больших 33.2 и кратных четырем:", count)

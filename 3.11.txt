p = int(input("Введите количество прошедших месяцев: "))
d = int(input("Введите количество дней после последнего месяца: "))

# создаем список со значениями количества дней в каждом месяце
days_in_month = [31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31]

# вычисляем общее количество дней с начала 1990 года
total_days = p * 30 + d + 2  # добавляем 2 дня, т.к. считаем с 1 января 1990 г.

# определяем месяц
month = 1
days_passed = 0
while days_passed < total_days:
    days_passed += days_in_month[month-1]
    month += 1

month -= 1  # последний месяц был пересчитан лишний раз
print("Данный день приходится на", month, "месяц")

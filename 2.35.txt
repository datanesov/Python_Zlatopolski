# Ввод данных
V1 = int(input("Введите скорость первого автомобиля (км/ч): "))
V2 = int(input("Введите скорость второго автомобиля (км/ч): "))
S = int(input("Введите расстояние между автомобилями (км): "))

# Расчет расстояния через 30 минут после опережения
t = 0.5 # 30 минут = 0.5 часа
D = S + (V1 - V2) * t

# Вывод результата
print("Расстояние между автомобилями через 30 минут после опережения первым автомобилем:", D, "км")

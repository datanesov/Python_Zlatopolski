
s = int(input("Введите значение s: "))
array = [100, 200, 300, 400, 500, 600, 700, 800, 900, 1000, 1100, 1200, 1300, 1400, 1500, 1600, 1700, 1800, 1900, 2000, 2100, 2200, 2300, 2400, 2500, 2600, 2700, 2800, 2900, 3000, 3100]

count = 0
for i in range(len(array)):
    if array[i] > s:
        count += 1

print("Количество дней, в которые стоимость проданных товаров превысила значение s:", count)

month_list = ["январь", "февраль", "март", "апрель", "май", "июнь",
              "июль", "август", "сентябрь", "октябрь", "ноябрь", "декабрь"]

month_num = int(input("Введите номер месяца: "))
if 1 <= month_num <= 12:
    print(month_list[month_num - 1])
else:
    print("Некорректный номер месяца!")

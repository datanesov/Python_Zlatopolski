
n = int(input("Введите количество элементов в массиве: "))
a = []
for i in range(n):
    a.append(int(input(f"Введите {i+1}-й элемент массива: ")))
    
s = int(input("Введите номер элемента, который нужно переместить: "))
k = int(input("Введите номер элемента, на место которого нужно переместить: "))

temp = a[s-1]
for i in range(s-1, k, -1):
    a[i] = a[i-1]
a[k] = temp

print("Новый массив:", a)


t = float(input("Введите время в минутах, прошедшее с начала очередного часа: "))
color = "зеленый"

while t >= 3:
    if color == "зеленый":
        color = "желтый"
        t -= 1
    elif color == "желтый":
        color = "красный"
        t -= 2
    elif color == "красный":
        color = "зеленый"
        t -= 3

if t < 3:
    print("Зеленый сигнал для водителей")
elif t < 4:
    print("Желтый сигнал для водителей")
elif t < 6:
    print("Красный сигнал для водителей")

arr = [10, 5, 15, 7, 15, 12, 20, 20]
max_elem = arr[0]
min_elem = arr[0]
max_count = 0
min_count = 0
for i in arr:
    if i > max_elem:
        max_elem = i
    elif i < min_elem:
        min_elem = i

for i in arr:
    if i == max_elem:
        max_count += 1
    elif i == min_elem:
        min_count += 1
print("Количество максимальных элементов:", max_count)
print("Количество минимальных элементов:", min_count)
